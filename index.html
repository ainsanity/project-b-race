function doGet() {
  var ss = SpreadsheetApp.openById('1lQiE-GpC_8Xy0VWiCDAffF9Lsw-YYB5tblxNFD24cbU');
  var sheet = ss.getSheets()[0];
  var data = sheet.getDataRange().getValues();
  var csv = data.map(r => r.map(c => '"'+String(c).replace(/"/g,'""')+'"').join(',')).join('\n');
  return ContentService.createTextOutput(JSON.stringify({csv:csv}))
    .setMimeType(ContentService.MimeType.JSON);
}
